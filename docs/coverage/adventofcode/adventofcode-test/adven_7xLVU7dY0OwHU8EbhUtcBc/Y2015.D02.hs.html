   10 module Y2015.D02
   11     ( Present(..)
   12     , parsePresents
   13     , surfaceArea
   14     , ribbonLength
   15 ) where
   16 
   17 import           Data.List          (foldl', sort)
   18 import qualified Text.Parsec as     P
   19 import           Text.Parsec.Char   (char, digit, endOfLine)
   20 import           Text.Parsec.String (Parser)
   21 
   22 import           Y2015.Util         (regularParse, intParser)
   23 
   24 -- |Represents a present in three dimensions
   25 data Present = Present Int Int Int
   26              deriving (Show, Eq)
   27 
   28 presentsParser :: Parser [Present]
   29 presentsParser = P.many1 (presentParser <* P.optional endOfLine)
   30 
   31 presentParser :: Parser Present
   32 presentParser = Present <$> (intParser <* char 'x')
   33                         <*> (intParser <* char 'x')
   34                         <*>  intParser
   35 
   36 -- |Parse presents from an input string
   37 parsePresents :: String          -- ^ Raw input of present dimensions
   38               -> Maybe [Present] -- ^ Possible list of 'Present's
   39 parsePresents s = case regularParse presentsParser s of
   40                        Right ps -> Just ps
   41                        Left _   -> Nothing
   42 
   43 -- |Find total surface area from list of 'Present's
   44 surfaceArea :: [Present] -- ^ List of 'Present's
   45             -> Int       -- ^ Total surface area of all 'Present's
   46 surfaceArea = foldl' (+) 0 . map wrapping
   47 
   48 wrapping :: Present -> Int
   49 wrapping p@(Present l w h) = sqft p + product (smallest p)
   50 
   51 smallest :: Present -> [Int]
   52 smallest (Present l w h) = take 2 $ sort [l, w, h]
   53 
   54 sqft :: Present -> Int
   55 sqft (Present l w h) = 2*l*w + 2*w*h + 2*h*l
   56 
   57 area :: Present -> Int
   58 area (Present l w h) = l*w*h
   59 
   60 -- |Find required length of ribbon for a list of presents.
   61 ribbonLength :: [Present] -- ^ List of 'Present's
   62              -> Int       -- ^ Total length of required ribbon
   63 ribbonLength = sum . map presentRibbon
   64 
   65 presentRibbon :: Present -> Int
   66 presentRibbon p = sum (map (*2) $ smallest p) + area p
   67 
   68 main :: IO ()
   69 main = do
   70        input <- readFile "src/Y2015/D02_input"
   71        case regularParse presentsParser input of
   72             Right presents -> do
   73                putStr "Part A: total square feet needed: "
   74                print (surfaceArea presents)
   75                putStr "Part B: total feet of ribbon needed: "
   76                print (ribbonLength presents)
   77             Left _         -> putStrLn "Error: Malformed input."
