   32 type Point = (Int, Int)
   33 
   34 -- |Represents a two-dimensional range of lights.
   35 data Range = Range Point Point
   36            deriving (Eq, Show)
   37 
   38 -- |Type of light grid instruction.
   39 data Instruction = On Range
   40                  | Off Range
   41                  | Toggle Range
   42                  deriving (Show)
   43 
   44 size :: Int
   45 size = 1000
   46 
   47 initialGrid :: R.Array R.U R.DIM2 Int
   48 initialGrid = R.fromListUnboxed
   49                   (Z :. size :. size :: R.DIM2)
   50                   (replicate (size*size) 0)
   51 
   52 instructionsParser :: Parser [Instruction]
   53 instructionsParser = P.many (instruction <* P.optional endOfLine)
   54 
   55 instruction :: Parser Instruction
   56 instruction = On     <$> directive "turn on"
   57           <|> Off    <$> directive "turn off"
   58           <|> Toggle <$> directive "toggle"
   59 
   60 directive :: String -> Parser Range
   61 directive s = P.skipMany1 (P.try (P.string s *> P.skipMany1 P.space)) *> range
   62 
   63 range :: Parser Range
   64 range = Range <$> point <* P.string " through " <*> point
   65 
   66 point :: Parser Point
   67 point = (,) <$> intParser <* char ',' <*> intParser
   68 
   69 -- |Folding function to aggregate computation for 'Instruction's per part
   70 -- |A spec.
   71 configureGridA :: R.Array R.U R.DIM2 Int -- ^ Light grid.
   72                -> Instruction            -- ^ Operation 'Instruction'.
   73                -> R.Array R.U R.DIM2 Int -- ^ Resultant light grid.
   74 configureGridA a (On range)     = switch a (const 1) range
   75 configureGridA a (Off range)    = switch a (const 0) range
   76 configureGridA a (Toggle range) = switch a toggle    range
   77 
   78 -- |Folding function to aggregate computation for 'Instruction's per part
   79 -- |B spec.
   80 configureGridB :: R.Array R.U R.DIM2 Int -- ^ Light grid.
   81                -> Instruction            -- ^ Operation 'Instruction'.
   82                -> R.Array R.U R.DIM2 Int -- ^ Resultant light grid.
   83 configureGridB a (On range)     = switch a (+1) range
   84 configureGridB a (Off range)    = switch a dim  range
   85 configureGridB a (Toggle range) = switch a (+2) range
   86 
   87 toggle :: Int -> Int
   88 toggle 1 = 0
   89 toggle _ = 1
   90 
   91 dim :: Int -> Int
   92 dim = max 0 . subtract 1
   93 
   94 switch :: (R.Source r a, Unbox a)
   95        => R.Array r R.DIM2 a
   96        -> (a -> a)
   97        -> Range
   98        -> R.Array R.U R.DIM2 a
   99 switch a f r = R.computeS $ R.traverse a id (set f r)
  100 
  101 -- This is pretty confusing:
  102 --    Custom mapping function (set the lights)
  103 -- -> Range to apply the function upon
  104 -- -> Function to retrieve original elements from
  105 -- -> Original array constructor
  106 -- -> New (or unchanged) value
  107 set :: (a -> a) -> Range -> (R.DIM2 -> a) -> R.DIM2 -> a
  108 set f (Range (x',y') (x'',y'')) g (Z :. x :. y)
  109     | withinX && withinY = f orig
  110     | otherwise          = orig
  111     where withinX = x >= x' && x <= x''
  112           withinY = y >= y' && y <= y''
  113           orig    = g (Z :. x :. y)
  114 
  115 -- |Execute 'Instruction' and return number of lit lights per part A spec.
  116 testA :: Instruction -- ^ Given 'Instruction'.
  117       -> Int         -- ^ Number of lit lights.
  118 testA = R.foldAllS (+) 0 . configureGridA initialGrid
  119 
  120 -- |Execute 'Instruction' and return number of lit lights per part B spec.
  121 testB :: Instruction -- ^ Given 'Instruction'
  122       -> Int         -- ^ Number of lit lights.
  123 testB = R.foldAllS (+) 0 . configureGridB initialGrid
  124 
  125 -- |Parses a string into a list of 'Instruction's.
  126 parseInstructions :: String                            -- ^ Input string to parse.
  127                   -> Either P.ParseError [Instruction] -- ^ Either an error or parsed structure.
  128 parseInstructions = regularParse instructionsParser
  129 
  130 -- |Run a light simulation
  131 lightSimulation :: (Monad m, Foldable t)
  132                 => (R.Array R.U R.DIM2 Int -> a -> R.Array R.U R.DIM2 Int) -- ^ REPA Light grid
  133                 -> t a                                                     -- ^ 'Instruction's
  134                 -> m Int                                                   -- ^ Lit lights
  135 lightSimulation f = R.sumAllP . foldl' f initialGrid
