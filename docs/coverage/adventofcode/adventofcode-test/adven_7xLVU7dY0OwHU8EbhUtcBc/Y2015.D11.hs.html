   10 module Y2015.D11 (rotate) where
   11 
   12 import Y2015.Util ((<&&>))
   13 import Data.List  (elemIndex, foldr, group, isInfixOf, iterate, tails)
   14 
   15 alphabet :: String
   16 alphabet = ['a'..'z']
   17 
   18 meetsReqs :: String -> Bool
   19 meetsReqs = hasPairs <&&> (not . forbidden) <&&> hasStraightFast
   20 
   21 hasPairs :: String -> Bool
   22 hasPairs = (1 <) . length . filter ((<) 1 . length) . group
   23 
   24 forbidden :: String -> Bool
   25 forbidden = any (`elem` "iol")
   26 
   27 hasStraightFast :: String -> Bool
   28 hasStraightFast = not . null . filterAsc . subSeqs
   29     where filterAsc = filter (`isInfixOf` alphabet)
   30           subSeqs   = takeWhile ((== 3) . length) . map (take 3) . tails
   31 
   32 hasStraightSlow :: String -> Bool
   33 hasStraightSlow []                          = False
   34 hasStraightSlow [x]                         = False
   35 hasStraightSlow [x,y]                       = False
   36 hasStraightSlow (x:y:z:zs) | straight x y z = True
   37                            | otherwise      = hasStraightSlow (y:z:zs)
   38                            where straight a b c = b == succ a && c == succ b
   39 
   40 increment :: String -> String
   41 increment = reverse . step . reverse
   42     where step []                 = []
   43           step [x]    | x == 'z'  = "aa"
   44                       | otherwise = [succ x]
   45           step (x:xs) | x /= 'z'  = succ x :      xs
   46                       | otherwise = 'a'    : step xs
   47 
   48 -- |Rotate a password within specific rules
   49 rotate :: String -- ^ Starting password
   50        -> String -- ^ Next valid password
   51 rotate = nextValid . increment
   52     where nextValid = head . filter meetsReqs . iterate increment
