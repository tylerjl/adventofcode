   10 module Y2015.D03 (santaRun, roboRun) where
   11 
   12 import           Data.List   (foldl')
   13 import           Data.Set    (Set)
   14 import qualified Data.Set as Set
   15 
   16 type Point = (Int, Int)
   17 
   18 direction :: Char -> Point
   19 direction c | c == '^'  = (0,  1)
   20             | c == 'v'  = (0, -1)
   21             | c == '>'  = (1,  0)
   22             | c == '<'  = (-1, 0)
   23             | otherwise = (0,  0)
   24 
   25 start :: Set Point
   26 start = Set.singleton (0, 0)
   27 
   28 move :: Point -> Point -> Point
   29 move (dx, dy) (x, y) = (x + dx, y + dy)
   30 
   31 -- |Find number of deliverables for santa's route
   32 santaRun :: String -- ^ Route input
   33          -> Int    -- ^ Number of stops
   34 santaRun = Set.size . deliver start . map direction
   35 
   36 -- |Find number of deliverables for the robot's route
   37 roboRun :: String -- ^ Route input
   38         -> Int    -- ^ Number of stops
   39 roboRun = Set.size . teamDelivery . tMap direction . divideWork
   40     where teamDelivery = uncurry (deliver . deliver start)
   41 
   42 tMap :: (a -> b) -> ([a], [a]) -> ([b], [b])
   43 tMap f (a1, a2) = (map f a1, map f a2)
   44 
   45 divideWork :: String -> (String, String)
   46 divideWork []  = ([],  [])
   47 divideWork [x] = ([x], [])
   48 divideWork (x:y:zs) = (x:xp, y:yp)
   49                       where (xp, yp) = divideWork zs
   50 
   51 deliver :: Set Point -> [Point] -> Set Point
   52 deliver = navigate (0, 0)
   53 
   54 navigate :: Point -> Set Point -> [Point] -> Set Point
   55 navigate origin history []          = history
   56 navigate origin history (dir:plans) =
   57         let newPoint = move dir origin
   58             step     = Set.insert newPoint history
   59         in  navigate newPoint step plans
