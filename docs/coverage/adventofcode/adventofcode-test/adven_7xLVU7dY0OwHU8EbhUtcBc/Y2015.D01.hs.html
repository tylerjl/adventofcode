   10 module Y2015.D01
   11     ( level
   12     , basement )
   13 where
   14 
   15 import Data.List (foldl')
   16 
   17 move :: Char -> Int
   18 move c | c == '('  = 1
   19        | c == ')'  = -1
   20        | otherwise = 0
   21 
   22 -- |Find final level from list of elevator movements
   23 level :: String -- ^ List of input open/close parens
   24       -> Int    -- ^ Final elevator level
   25 level = foldl' (+) 0 . map move
   26 
   27 -- |Find position that arrives at level 0
   28 basement :: String    -- ^ List of input open/close parens
   29          -> Maybe Int -- ^ Possible position in string that arrives at zero
   30 basement = find 0 1 . map move
   31     where find current idx (move:moves)
   32                | current + move < 0 = Just idx
   33                | otherwise          = find (current + move) (idx + 1) moves
   34           find _ _ [] = Nothing
