   14 import Data.List (foldl')
   15 import qualified Data.Set as Set
   16 
   17 data Direction = North
   18                | East
   19                | South
   20                | West
   21                deriving (Enum)
   22 
   23 type Coordinates = (Int, Int)
   24 
   25 blockDistance :: String -> Int
   26 blockDistance = blocks
   27               . foldl' travel (0, 0)
   28               . mapRoute
   29 
   30 visitedTwice :: String -> Maybe Int
   31 visitedTwice = navigate (0,0) Set.empty
   32              . mapRoute
   33     where navigate point set (path:xs)
   34               | Set.member point' set = Just $ blocks point'
   35               | otherwise = navigate point' (Set.insert point' set) xs
   36               where point' = travel point path
   37           navigate _ _ _ = Nothing
   38 
   39 mapRoute :: String -> [Direction]
   40 mapRoute = toRoute North
   41          . map (toPath . filter  (not . (==) ','))
   42          . words
   43          where toRoute orientation ((lr, distance):xs) =
   44                    replicate distance orientation' ++ toRoute orientation' xs
   45                    where orientation' = turn lr orientation
   46                toRoute _ [] = []
   47 
   48 blocks :: Coordinates -> Int
   49 blocks (x, y) = abs x + abs y
   50 
   51 travel :: Coordinates -> Direction -> Coordinates
   52 travel (x, y) d = case d of
   53                     North -> (x, succ y)
   54                     East -> (succ x, y)
   55                     South -> (x, pred y)
   56                     West -> (pred x, y)
   57 
   58 toPath :: String -> (Char, Int)
   59 toPath (direction:steps) = (direction, read steps)
   60 toPath [] = (' ', 0)
   61 
   62 turn :: Char -> Direction -> Direction
   63 turn 'R' West = North
   64 turn 'L' North = West
   65 turn 'R' d = succ d
   66 turn 'L' d = pred d
   67 turn _ d = d
